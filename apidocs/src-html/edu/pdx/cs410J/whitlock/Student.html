001package edu.pdx.cs410J.whitlock;
002
003import edu.pdx.cs410J.lang.Human;
004
005import java.util.ArrayList;
006import java.util.List;
007
008/**                                                                                 
009 * This class is represents a <code>Student</code>.                                 
010 */                                                                                 
011public class Student extends Human {
012
013  static final String MALE = "male";
014  static final String FEMALE = "female";
015  private final String gender;
016  private final double gpa;
017  private final List<String> classes;
018
019  /**
020   * Creates a new <code>Student</code>                                             
021   *                                                                                
022   * @param name                                                                    
023   *        The student's name                                                      
024   * @param classes                                                                 
025   *        The names of the classes the student is taking.  A student              
026   *        may take zero or more classes.                                          
027   * @param gpa                                                                     
028   *        The student's grade point average                                       
029   * @param gender                                                                  
030   *        The student's gender ("male" or "female", case insensitive)             
031   */                                                                               
032  public Student(String name, List<String> classes, double gpa, String gender) {
033    super(name);
034
035    this.gender = gender;
036    this.gpa = gpa;
037    this.classes = classes;
038  }
039
040  /**                                                                               
041   * All students say "This class is too much work"
042   */
043  @Override
044  public String says() {                                                            
045    return "This class is too much work";
046  }
047                                                                                    
048  /**                                                                               
049   * Returns a <code>String</code> that describes this                              
050   * <code>Student</code>.                                                          
051   */                                                                               
052  public String toString() {
053    return this.name + " has a GPA of " + this.gpa + " and is taking " +
054      getNumberOfClasses() + ".  " +
055      getGenderPronoun() + " says \"" + says() + "\".";
056  }
057
058  private String getNumberOfClasses() {
059    if (this.classes.isEmpty()) {
060      return "no classes";
061
062    } else if (this.classes.size() == 1) {
063      return "1 class: " + this.classes.get(0);
064
065    } else {
066      String phrase = this.classes.size() + " classes: ";
067
068      if (this.classes.size() == 2) {
069        phrase += this.classes.get(0) + " and " + this.classes.get(1);
070
071      } else {
072        for (int i = 0; i < classes.size() - 1; i++) {
073          String className = classes.get(i);
074          phrase += className;
075          phrase += ", ";
076        }
077
078        phrase += "and " + classes.get(classes.size() - 1);
079      }
080
081      return phrase;
082    }
083  }
084
085  private String getGenderPronoun() {
086    switch (this.gender) {
087      case MALE:
088        return "He";
089
090      case FEMALE:
091        return "She";
092
093      default:
094        throw new UnknownGenderException(this.gender);
095    }
096  }
097
098  /**
099   * Main program that parses the command line, creates a
100   * <code>Student</code>, and prints a description of the student to
101   * standard out by invoking its <code>toString</code> method.
102   */
103  public static void main(String[] args) {
104    if (args.length == 0) {
105      printErrorMessageAndExit("Missing command line arguments");
106    }
107
108    String name = args[0];
109
110    if (args.length < 2) {
111      printErrorMessageAndExit("Missing gender");
112    }
113    String gender = args[1];
114    double gpa = parseGpa(args[2]);
115
116    List<String> classes = new ArrayList<>();
117    for (int i = 3; i < args.length; i++) {
118      classes.add(args[i]);
119    }
120
121    Student student = new Student(name, classes, gpa, gender);
122    try {
123      System.out.println(student.toString());
124    } catch (UnknownGenderException ex) {
125      printErrorMessageAndExit("Unknown gender: " + ex.getUnknownGender());
126    }
127
128    System.exit(0);
129  }
130
131  private static void printErrorMessageAndExit(String message) {
132    System.err.println(message);
133    System.exit(1);
134  }
135
136  private static double parseGpa(String gpa) {
137    try {
138      return Double.parseDouble(gpa);
139
140    } catch (NumberFormatException ex) {
141      printErrorMessageAndExit("Invalid GPA: " + gpa);
142      return Double.NaN;
143    }
144  }
145}
