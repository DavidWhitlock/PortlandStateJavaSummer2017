16      @Koan
17      public void localizedOutputOfDates() {
18          Calendar cal = Calendar.getInstance();
19          cal.set(2011, 3, 3);
20          Date date = cal.getTime();
21          Locale localeBR = new Locale("pt", "BR"); // portuguese, Brazil
22          DateFormat dateformatBR = DateFormat.getDateInstance(DateFormat.FULL, localeBR);
23          assertEquals(dateformatBR.format(date), __);
24  
25          Locale localeJA = new Locale("de"); // German
26          DateFormat dateformatJA = DateFormat.getDateInstance(DateFormat.FULL, localeJA);
27          // Well if you don't know how to type these characters, try "de", "it" or "us" ;-)
28          assertEquals(dateformatJA.format(date), __);
29      }
30  
31      @Koan
32      public void getCountryInformation() {
33          Locale locBR = new Locale("pt", "BR");
34          assertEquals(locBR.getDisplayCountry(), __);
35          assertEquals(locBR.getDisplayCountry(locBR), __);
36  
37          Locale locCH = new Locale("it", "CH");
38          assertEquals(locCH.getDisplayCountry(), __);
39          assertEquals(locCH.getDisplayCountry(locCH), __);
40          assertEquals(locCH.getDisplayCountry(new Locale("de", "CH")), __);
41      }
